@inject ToastManager ToastManager

    <FluentButton Appearance=Appearance.Neutral @onclick="ClearAllPreserveQueue">Clear all, preserve queue</FluentButton>

    <FluentButton Appearance=Appearance.Neutral @onclick="ClearCustom">Clear custom </FluentButton>

    <FluentButton Appearance=Appearance.Neutral @onclick="ClearWarning">
        <FluentIcon Name="@FluentIcons.Warning" Size="@IconSize.Size20" Variant="@IconVariant.Filled" Color="@Color.Warning" Slot="start" />
        Clear warning
    </FluentButton>

    <FluentButton Appearance=Appearance.Neutral @onclick="ClearError">
        <FluentIcon Name="@FluentIcons.DismissCircle" Size="@IconSize.Size20" Variant="@IconVariant.Filled" Color="@Color.Error" Slot="start" />
        Clear error
    </FluentButton>

    <FluentButton Appearance=Appearance.Neutral @onclick="ClearInfo">
        <FluentIcon Name="@FluentIcons.Info" Size="@IconSize.Size20" Variant="@IconVariant.Filled" Color="@Color.Info" Slot="start" />
        Clear info
    </FluentButton>

    <FluentButton Appearance=Appearance.Neutral @onclick="ClearProgress">
        <FluentIcon Name="@FluentIcons.Flash" Size="@IconSize.Size20" Color="@Color.Neutral" Slot="start" />
        Clear progress
    </FluentButton>

    <FluentButton Appearance=Appearance.Neutral @onclick="ClearUpload">
        <FluentIcon Name="@FluentIcons.ArrowUpload" Size="@IconSize.Size20" Color="@Color.Neutral" Slot="start" />

    </FluentButton>

    <FluentButton Appearance=Appearance.Neutral @onclick="ClearDownload">
        <FluentIcon Name="@FluentIcons.ArrowDownload" Size="@IconSize.Size20" Color="@Color.Neutral" Slot="start" />

        <FluentIcon Name="@FluentIcons.Delete" Size="@IconSize.Size20" Color="@Color.Neutral" Slot="start" />
        Clear custom
    </FluentButton>
</FluentStack>

@code {
    private void ClearAll()
        => ToastManager.ClearAll();

    private void ClearAllPreserveQueue()
        => ToastManager.ClearAll(false);

    private void ClearCustom()
        => ToastManager.ClearCustomToasts();

    private void ClearIntent()
        => ToastManager.ClearToasts(ToastIntent.Warning);

    private void ClearSuccess()
        => ToastManager.ClearSuccessToasts();

    private void ClearWarning()
        => ToastManager.ClearWarningToasts();

    private void ClearError()
        => ToastManager.ClearErrorToasts();

    private void ClearInfo()
        => ToastManager.ClearInfoToasts();

    private void ClearProgress()
       => ToastManager.ClearProgressToasts();

    private void ClearUpload()
        => ToastManager.ClearUploadToasts();

    private void ClearDownload()
       => ToastManager.ClearDownloadToasts();

    private void ClearEvent()
        => ToastManager.ClearEventToasts();

    private void ClearAvatar()
       => ToastManager.ClearAvatarToasts();

    private void ClearCustomIntent()
        => ToastManager.ClearCustomIntentToasts();
}
