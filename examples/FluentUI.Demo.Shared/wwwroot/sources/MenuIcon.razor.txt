@* 
    We can't use the FluentIcon's own Slot parameter here because it will not be detected by FluentMenu's internal script to determine 
    indentation (they are added too late in the render process). Workaround is to define a slot manually and place the FluentIcon in there
*@
<FluentMenu Open="true">
    <FluentMenuItem>
        Menu item 1
        <span slot="start" >
            <FluentIcon Name="@FluentIcons.Globe" Size="@IconSize.Size16" Color="Color.Neutral" />
        </span>
        <span slot="end">
            <FluentIcon Name="@FluentIcons.Globe" Size="@IconSize.Size16" Color="Color.Neutral" />
        </span>
    </FluentMenuItem>
    <FluentMenuItem>
        Menu item 2
        <span slot="start">
            <FluentIcon Name="@FluentIcons.Globe" Size="@IconSize.Size16" Color="Color.Neutral" />
        </span>
        <span slot="end">
            <FluentIcon Name="@FluentIcons.Globe" Size="@IconSize.Size16" Color="Color.Neutral" />
        </span>
    </FluentMenuItem>
    <FluentMenuItem>
        Menu item 3
        <span slot="start">
            <FluentIcon Name="@FluentIcons.Globe" Size="@IconSize.Size16" Variant=IconVariant.Filled Color=Color.Accent />
        </span>
        <span slot="end">
            <FluentIcon Name="@FluentIcons.Globe" Size="@IconSize.Size16" Variant=IconVariant.Filled Color=Color.Accent />
        </span>
    </FluentMenuItem>
</FluentMenu>
