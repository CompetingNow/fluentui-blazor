@inject IToastService Toast


<p>Click on this button to open a Toast.</p>

<FluentButton @onclick="@OpenToast" Appearance="Appearance.Accent">Open</FluentButton>

@code
{
    int counter = 0;
    void OpenToast()
    {
        var intent = Enum.GetValues<ToastIntent>()[counter % 10];
        var message = $"Simple Toast #{counter++}";
        Toast.ShowToast(intent, message);
    }
}
